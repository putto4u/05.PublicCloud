include "root" {
  path = find_in_parent_folders()
}

# 라우팅에 필요한 기존 리소스(VPC, TGW)의 종속성(Dependency) 및 출력 값 확보
dependency "hub_vpc" { config_path = "../01-hub-vpc" }
dependency "service_vpc" { config_path = "../02-service-vpc" }
dependency "tgw" { config_path = "../03-tgw" }




# 순수 Terraform 코드를 동적으로 생성하여 TGW(Transit Gateway) 전용 라우팅 규칙 주입
generate "routes" {
  path      = "vpc_routes.tf"
  if_exists = "overwrite_terragrunt"
  contents  = <<EOF
variable "hub_public_route_tables" { type = list(string) }
variable "service_private_route_tables" { type = list(string) }
variable "tgw_id" { type = string }

# 1. Hub VPC (Public) -> Service VPC (10.1.0.0/16) 라우팅
# ALB(Application Load Balancer)가 Service VPC 내부의 백엔드로 트래픽을 전달하기 위한 경로
resource "aws_route" "hub_to_service" {
  count                  = length(var.hub_public_route_tables)
  route_table_id         = var.hub_public_route_tables[count.index]
  destination_cidr_block = "10.1.0.0/16"
  transit_gateway_id     = var.tgw_id
}

# 2. Service VPC (Private) -> 외부 인터넷 (0.0.0.0/0) 라우팅
# Service VPC의 프라이빗 인스턴스가 패치 업데이트 등을 위해 외부로 나갈 때 TGW를 거쳐 Hub VPC로 향하는 경로
resource "aws_route" "service_to_hub" {
  count                  = length(var.service_private_route_tables)
  route_table_id         = var.service_private_route_tables[count.index]
  destination_cidr_block = "0.0.0.0/0"
  transit_gateway_id     = var.tgw_id
}

# 생성된 라우팅 규칙의 ID를 출력하여 확인 가능하게 합니다.
output "hub_to_service_route_ids" {
  value = aws_route.hub_to_service[*].id
}

output "service_to_hub_route_ids" {
  value = aws_route.service_to_hub[*].id
}




EOF
}

# 동적 생성된 Terraform 코드의 변수에 매핑할 실제 값을 주입 (igw_id 제거됨)
inputs = {
  hub_public_route_tables      = dependency.hub_vpc.outputs.public_route_table_ids
  service_private_route_tables = dependency.service_vpc.outputs.private_route_table_ids
  tgw_id                       = dependency.tgw.outputs.ec2_transit_gateway_id
}
