## 우분투 테라폼 설치

1. sudo apt-get update && sudo apt-get install -y gnupg software-properties-common wget
   
   ## 패키지의 무결성을 검증하기 위해 공식 GPG 키를 다운로드하여 시스템의 키링(Keyring, 키 보관소)에 저장
3. wget -O- https://apt.releases.hashicorp.com/gpg | sudo gpg --dearmor -o /usr/share/keyrings/hashicorp-archive-keyring.gpg
4. 공식 리포지토리(Repository)
    echo "deb [signed-by=/usr/share/keyrings/hashicorp-archive-keyring.gpg] \
    https://apt.releases.hashicorp.com $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/hashicorp.list
    
5. sudo apt-get update && sudo apt-get install terraform
6. terraform -v
7. terraform -install-autocomplete  # 자동완성

   
## 테라그런트 설치하기

1. # 릴리즈 버전은 공식 깃허브를 참조하여 사내 표준 버전에 맞게 변경
export TG_VERSION="v0.55.0" 
wget "https://github.com/gruntwork-io/terragrunt/releases/download/${TG_VERSION}/terragrunt_linux_amd64" 

2. # 실행 권한 부여
chmod +x terragrunt_linux_amd64

# 시스템 환경 변수(PATH)에 포함된 디렉토리로 이동 및 이름 변경
sudo mv terragrunt_linux_amd64 /usr/local/bin/terragrunt
terragrunt --version
