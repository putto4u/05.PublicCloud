#!/bin/bash # 1. 시스템 패키지 업데이트 및 업그레이드 # -y 옵션을 통해 사용자 확인 절차 없이 자동 진행합니다. apt-get update -y apt-get upgrade -y # 2. Nginx(엔진엑스) 설치 apt-get install nginx -y # 3. Nginx 서비스 시작 및 부팅 시 자동 실행 설정 systemctl start nginx systemctl enable nginx # 4. 기본 웹 페이지 내용 수정 (설치 확인용) # 인스턴스의 호스트네임을 포함한 간단한 HTML 파일을 생성합니다. TOKEN=$(curl -X PUT "http://169.254.169.254/latest/api/token" -H "X-aws-ec2-metadata-token-ttl-seconds: 21600") INSTANCE_ID=$(curl -H "X-aws-ec2-metadata-token: $TOKEN" http://169.254.169.254/latest/meta-data/instance-id) cat < /var/www/html/index.html EC2 Nginx 자동 설치 완료

Nginx가 성공적으로 설치되었습니다!

이 웹 서버는 EC2 User Data 스크립트를 통해 자동으로 구성되었습니다.

인스턴스 ID: $INSTANCE_ID

EOF # 5. 변경사항 적용을 위한 Nginx 재시작 systemctl restart nginx
